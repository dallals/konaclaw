# kc-memory Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add the persistent memory layer so KonaClaw and every subagent remember things between conversations. Two memory tiers: a **global** `~/KonaClaw/memory/user.md` shared by all agents (the user's profile, preferences, ongoing projects), and a **per-agent** `~/KonaClaw/memory/<agent>/MEMORY.md`. Memory is loaded into every agent turn's prompt; memory writes go through kc-sandbox so they're journaled and undoable.

**Architecture:** A small Python package `kc_memory` with a `MemoryStore` that owns the directory layout, a `MemoryReader` that produces the prefix block injected into each system prompt at turn start, and three sandboxed tools — `memory.read`, `memory.append`, `memory.replace` — registered into every agent so the model can update its own memory mid-conversation. Writes hit kc-sandbox's `Journal` + `UndoLog` so any memory edit can be undone exactly like a file edit.

**Tech Stack:** Python 3.11+. Depends on `kc-core`, `kc-sandbox`. Tiny — no new third-party deps.

```python
# src/kc_memory/store.py
from __future__ import annotations
from pathlib import Path


class MemoryStore:
    USER_FILENAME = "user.md"
    AGENT_FILENAME = "MEMORY.md"

    def __init__(self, root: Path) -> None:
        self.root = Path(root)

    def init(self) -> None:
        self.root.mkdir(parents=True, exist_ok=True)
        if not self.user_path().exists():
            self.user_path().write_text("# User profile\n\n_(empty — agents may write here)_\n")

    def user_path(self) -> Path:
        return self.root / self.USER_FILENAME

    def agent_path(self, agent: str) -> Path:
        return self.root / agent / self.AGENT_FILENAME

    # ---- read ----

    def read_user(self) -> str:
        p = self.user_path()
        return p.read_text() if p.exists() else ""

    def read_agent(self, agent: str) -> str:
        p = self.agent_path(agent)
        return p.read_text() if p.exists() else ""

    # ---- write (non-journaled — for tests / direct use) ----

    def write_user(self, content: str) -> None:
        self.user_path().parent.mkdir(parents=True, exist_ok=True)
        self.user_path().write_text(content)

    def write_agent(self, agent: str, content: str) -> None:
        p = self.agent_path(agent)
        p.parent.mkdir(parents=True, exist_ok=True)
        p.write_text(content)

    def append_user(self, suffix: str) -> None:
        self.write_user(self.read_user() + suffix)

    def append_agent(self, agent: str, suffix: str) -> None:
        self.write_agent(agent, self.read_agent(agent) + suffix)
```

```python
# src/kc_memory/reader.py
from __future__ import annotations
from kc_memory.store import MemoryStore


class MemoryReader:
    def __init__(self, store: MemoryStore) -> None:
        self.store = store

    def format_prefix(self, agent: str) -> str:
        parts: list[str] = []

        user = self.store.read_user().strip()
        # Strip the auto-init placeholder so it doesn't pollute the prompt
        if user and "_(empty — agents may write here)_" not in user:
            parts.append("## User profile\n\n" + user)

        agent_mem = self.store.read_agent(agent).strip()
        if agent_mem:
            parts.append(f"## {agent} memory\n\n" + agent_mem)

        if not parts:
            return ""
        return (
            "<!-- KonaClaw memory — do not invent or hallucinate; update via memory tools -->\n\n"
            + "\n\n".join(parts)
            + "\n\n---\n\n"
        )
```

## Task 3: Memory Tools (Sandboxed, Journaled, Undoable)

**Files:**
- Create: `src/kc_memory/tools.py`
- Test: `tests/test_tools.py`

**Why:** `memory.read`, `memory.append`, `memory.replace` — three tools the agent can call to update its own memory. Writes go through `kc_sandbox.Journal` + `UndoLog` exactly like file writes, so any memory edit can be undone from the dashboard. Each tool takes a `scope` argument: `"user"` writes to `user.md`, `"agent"` writes to the agent's own `MEMORY.md`. An agent cannot edit another agent's memory — the `agent_name` is bound at tool-build time, not passed by the model.

```python
# src/kc_memory/tools.py
from __future__ import annotations
from pathlib import Path
from kc_core.tools import Tool
from kc_sandbox.journal import Journal
from kc_sandbox.undo import UndoLog, UndoEntry
from kc_memory.store import MemoryStore


def build_memory_tools(
    *,
    store: MemoryStore,
    journal: Journal,
    undo_log: UndoLog,
    agent_name: str,
) -> dict[str, Tool]:

    def _path_for(scope: str) -> Path:
        if scope == "user": return store.user_path()
        if scope == "agent": return store.agent_path(agent_name)
        raise ValueError(f"invalid scope {scope!r} (must be 'user' or 'agent')")

    def _commit(path: Path, action: str) -> str:
        sha = journal.commit(
            message=f"memory.{action} {path.relative_to(journal.root)}",
            author_agent=agent_name, paths=[path],
        )
        undo_log.record(UndoEntry(
            agent=agent_name, tool=f"memory.{action}",
            reverse_kind="git-revert",
            reverse_payload={"share": "memory", "sha": sha},
        ))
        return sha

    def memory_read(scope: str) -> str:
        if scope == "user": return store.read_user() or "(empty)"
        if scope == "agent": return store.read_agent(agent_name) or "(empty)"
        raise ValueError(f"invalid scope {scope!r}")

    def memory_append(scope: str, content: str) -> str:
        path = _path_for(scope)
        path.parent.mkdir(parents=True, exist_ok=True)
        existing = path.read_text() if path.exists() else ""
        path.write_text(existing + content)
        sha = _commit(path, "append")
        return f"appended {len(content)} chars to {scope} memory (commit {sha[:7]})"

    def memory_replace(scope: str, content: str) -> str:
        path = _path_for(scope)
        path.parent.mkdir(parents=True, exist_ok=True)
        path.write_text(content)
        sha = _commit(path, "replace")
        return f"replaced {scope} memory ({len(content)} chars, commit {sha[:7]})"

    def make(name, desc, params, impl):
        return Tool(name=name, description=desc, parameters=params, impl=impl)

    return {
        "memory.read":    make("memory.read", "Read your own memory or the global user profile.",
            {"type": "object", "properties": {"scope": {"type": "string", "enum": ["user", "agent"]}}, "required": ["scope"]}, memory_read),
        "memory.append":  make("memory.append", "Append text to your own memory or the global user profile. Journaled & undoable.",
            {"type": "object", "properties": {"scope": {"type": "string", "enum": ["user", "agent"]}, "content": {"type": "string"}}, "required": ["scope", "content"]}, memory_append),
        "memory.replace": make("memory.replace", "Replace your own memory or the global user profile entirely. Journaled & undoable.",
            {"type": "object", "properties": {"scope": {"type": "string", "enum": ["user", "agent"]}, "content": {"type": "string"}}, "required": ["scope", "content"]}, memory_replace),
    }
```

```python
# src/kc_memory/wiring.py
from __future__ import annotations
from kc_core.agent import Agent
from kc_sandbox.journal import Journal
from kc_sandbox.undo import UndoLog
from kc_memory.store import MemoryStore
from kc_memory.reader import MemoryReader
from kc_memory.tools import build_memory_tools


def attach_memory_to_agent(
    agent: Agent,
    *,
    store: MemoryStore,
    journal: Journal,
    undo_log: UndoLog,
) -> None:
    """Mutate the given Agent: prepend memory to system prompt + register memory tools."""
    reader = MemoryReader(store=store)
    prefix = reader.format_prefix(agent=agent.name)
    if prefix:
        agent.system_prompt = prefix + agent.system_prompt

    tools = build_memory_tools(
        store=store, journal=journal, undo_log=undo_log, agent_name=agent.name,
    )
    for t in tools.values():
        # Skip if already registered (idempotent)
        if t.name in agent.tools.names():
            continue
        agent.tools.register(t)
```
