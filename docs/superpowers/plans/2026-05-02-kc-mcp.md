# kc-mcp Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add MCP (Model Context Protocol) integration to KonaClaw. Static MCPs are loaded from config at supervisor boot. Agents can also install new MCP servers at runtime by calling an `install_mcp_server` meta-tool — gated by the destructive-tier approval flow built in kc-supervisor (Task 9). Newly installed MCP tools default to **destructive tier** until the user downgrades them.

**Architecture:** A separate Python package `kc_mcp` that depends on `kc-core`, `kc-sandbox`, and `kc-supervisor`. Built on the official Python [`mcp`](https://pypi.org/project/mcp/) SDK. A `MCPManager` owns the catalogue of running MCP servers, each represented by an `MCPServerHandle` that holds the SDK client + the spawned subprocess. Each MCP tool is wrapped as a `kc_core.tools.Tool` whose `impl` calls the MCP server. `mcp_installs` SQLite table tracks every install (added in this plan to `kc_supervisor.storage`).

**Tech Stack:** Python 3.11+, `mcp>=1.0` (Anthropic's official SDK), depends on prior sub-projects.

```python
# src/kc_mcp/store.py
from __future__ import annotations
import json
import time
from dataclasses import dataclass, field
from typing import Optional
from kc_supervisor.storage import Storage


SCHEMA_EXTENSION = """
CREATE TABLE IF NOT EXISTS mcp_installs (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL UNIQUE,
    command TEXT NOT NULL,
    args_json TEXT NOT NULL,
    env_json TEXT NOT NULL,
    why TEXT,
    installed_by_agent TEXT NOT NULL,
    ts REAL NOT NULL,
    status TEXT NOT NULL DEFAULT 'running'
);
"""


@dataclass
class MCPInstallRow:
    name: str
    command: str
    args: list[str]
    env: dict[str, str]
    why: str
    installed_by_agent: str
    id: Optional[int] = None
    ts: float = field(default_factory=time.time)
    status: str = "running"


class MCPInstallStore:
    def __init__(self, storage: Storage) -> None:
        self.s = storage

    def init(self) -> None:
        with self.s.connect() as c:
            c.executescript(SCHEMA_EXTENSION)

    def register(self, row: MCPInstallRow) -> int:
        with self.s.connect() as c:
            cur = c.execute(
                "INSERT INTO mcp_installs (name, command, args_json, env_json, why, installed_by_agent, ts, status) "
                "VALUES (?,?,?,?,?,?,?,?)",
                (row.name, row.command, json.dumps(row.args), json.dumps(row.env),
                 row.why, row.installed_by_agent, row.ts, row.status),
            )
            return int(cur.lastrowid)

    def list(self) -> list[dict]:
        with self.s.connect() as c:
            rows = c.execute("SELECT * FROM mcp_installs ORDER BY ts DESC").fetchall()
        out = []
        for r in rows:
            d = dict(r)
            d["args"] = json.loads(d.pop("args_json"))
            d["env"] = json.loads(d.pop("env_json"))
            out.append(d)
        return out

    def set_status(self, install_id: int, status: str) -> None:
        with self.s.connect() as c:
            c.execute("UPDATE mcp_installs SET status=? WHERE id=?", (status, install_id))
```

```python
# src/kc_mcp/handle.py
from __future__ import annotations
from dataclasses import dataclass, field
from typing import Any, Optional


@dataclass
class MCPTool:
    name: str
    description: str
    inputSchema: dict[str, Any]


class MCPServerHandle:
    """Owns one running MCP server: subprocess + SDK client."""

    def __init__(self, name: str, command: str, args: list[str], env: dict[str, str]) -> None:
        self.name = name
        self.command = command
        self.args = args
        self.env = env
        self._client: Any = None    # injected for tests; real start() sets via SDK
        self._proc: Any = None

    async def start(self) -> None:
        """Spawn the subprocess and connect via mcp.stdio_client."""
        from mcp import ClientSession, StdioServerParameters
        from mcp.client.stdio import stdio_client
        params = StdioServerParameters(command=self.command, args=self.args, env=self.env)
        self._reader_writer_cm = stdio_client(params)
        read, write = await self._reader_writer_cm.__aenter__()
        self._session_cm = ClientSession(read, write)
        self._client = await self._session_cm.__aenter__()
        await self._client.initialize()

    async def list_tools(self) -> list[MCPTool]:
        result = await self._client.list_tools()
        # The official SDK returns a ListToolsResult with .tools — we normalize:
        items = getattr(result, "tools", result)
        return [MCPTool(name=t.name, description=t.description or "", inputSchema=t.inputSchema or {})
                for t in items]

    async def call_tool(self, name: str, arguments: dict) -> str:
        result = await self._client.call_tool(name, arguments)
        # Normalize: MCP returns CallToolResult with .content list of TextContent
        content = getattr(result, "content", None)
        if content and hasattr(content[0], "text"):
            return "".join(getattr(c, "text", "") for c in content)
        return str(result)

    async def shutdown(self) -> None:
        try:
            if hasattr(self, "_session_cm"):
                await self._session_cm.__aexit__(None, None, None)
            if hasattr(self, "_reader_writer_cm"):
                await self._reader_writer_cm.__aexit__(None, None, None)
        except Exception:
            pass
```

```python
# src/kc_mcp/tool_adapter.py
from __future__ import annotations
from kc_core.tools import Tool
from kc_mcp.handle import MCPServerHandle, MCPTool


def mcp_tool_to_kc_tool(handle: MCPServerHandle, mcp_tool: MCPTool) -> Tool:
    name = f"mcp.{handle.name}.{mcp_tool.name}"

    async def impl(**kwargs):
        return await handle.call_tool(mcp_tool.name, kwargs)

    return Tool(
        name=name,
        description=mcp_tool.description,
        parameters=mcp_tool.inputSchema or {"type": "object"},
        impl=impl,
    )
```

(Note: `Tool.impl` was originally typed as a sync callable, but kc-core's tool execution path doesn't actually invoke it sync — the agent loop calls `self.tools.invoke(...)`, which is sync, then `str(result)`s it. To support async tool impls we add a tiny adapter in kc-core's `ToolRegistry.invoke`.)

```python
# src/kc_mcp/manager.py
from __future__ import annotations
from kc_core.tools import Tool
from kc_mcp.handle import MCPServerHandle
from kc_mcp.tool_adapter import mcp_tool_to_kc_tool


class MCPManager:
    def __init__(self) -> None:
        self._handles: dict[str, MCPServerHandle] = {}
        self._tools_by_handle: dict[str, list[Tool]] = {}

    async def register_handle(self, handle: MCPServerHandle) -> list[Tool]:
        if handle.name in self._handles:
            raise ValueError(f"MCP server {handle.name!r} already registered")
        await handle.start()
        mcp_tools = await handle.list_tools()
        wrapped = [mcp_tool_to_kc_tool(handle, t) for t in mcp_tools]
        self._handles[handle.name] = handle
        self._tools_by_handle[handle.name] = wrapped
        return wrapped

    async def unregister(self, name: str) -> None:
        if name not in self._handles:
            return
        h = self._handles.pop(name)
        self._tools_by_handle.pop(name, None)
        await h.shutdown()

    def all_tools(self) -> list[Tool]:
        out: list[Tool] = []
        for ts in self._tools_by_handle.values():
            out.extend(ts)
        return out

    def names(self) -> list[str]:
        return list(self._handles.keys())
```

```python
# src/kc_mcp/meta_tool.py
from __future__ import annotations
from typing import Any, Callable
from kc_core.tools import Tool
from kc_mcp.manager import MCPManager
from kc_mcp.handle import MCPServerHandle
from kc_mcp.store import MCPInstallStore, MCPInstallRow
from kc_supervisor.approvals import ApprovalBroker


def build_install_mcp_server_tool(
    *,
    manager: MCPManager,
    store: MCPInstallStore,
    broker: ApprovalBroker,
    agent_name: str,
    handle_factory: Callable[..., MCPServerHandle] = MCPServerHandle,
) -> Tool:
    """The meta-tool agents call to request a new MCP server install."""

    async def impl(name: str, command: str, args: list[str], env: dict[str, str], why: str = "") -> str:
        # Destructive-tier approval, fired through the supervisor's broker so
        # it pops up in the dashboard's Permissions queue exactly like any
        # other destructive action.
        allowed, reason = await broker.request_approval(
            agent=agent_name,
            tool="install_mcp_server",
            arguments={"name": name, "command": command, "args": args, "env": list(env.keys()), "why": why},
        )
        if not allowed:
            return f"install denied: {reason or 'user said no'}"

        try:
            handle = handle_factory(name=name, command=command, args=args, env=env)
            await manager.register_handle(handle)
        except Exception as e:
            return f"install failed: {type(e).__name__}: {e}"

        store.register(MCPInstallRow(
            name=name, command=command, args=list(args), env=dict(env),
            why=why, installed_by_agent=agent_name,
        ))
        new_tools = [t.name for t in manager.all_tools() if t.name.startswith(f"mcp.{name}.")]
        return f"installed {name} ({len(new_tools)} tools): {', '.join(new_tools)}"

    return Tool(
        name="install_mcp_server",
        description="Install a new MCP server to gain access to its tools. Requires user approval.",
        parameters={
            "type": "object",
            "properties": {
                "name":    {"type": "string", "description": "Short identifier for the server"},
```
